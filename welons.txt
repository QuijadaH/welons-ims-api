--- Order tables ---

	Order
	- Control no., PK VARCHAR NOT NULL    # Derive from contact person, site location, and creation date (<contact person first initial><site location first initial>-<yymmdd>)
	- Project code, VARCHAR NOT NULL
	- Project details, Text NULL
	- Site location, Text NULL
	- Subtotal, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Other works subtotal, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Overhead cost, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Delivery fee, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Grand total, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Date fully paid, Date NULL
	- Unit status ID, FK
	- Order status ID, FK
	- Dispatch date, Date NULL
	- Lead source ID, FK
	- Catered by ID, FK

	Unit status
	- ID, PK unsigned Int
	- Unit status, VARCHAR NOT NULL UNIQUE

	Order status
	- ID, PK unsigned Int
	- Order status, VARCHAR NOT NULL UNIQUE

	Lead source
	- ID, PK unsigned Int
	- Lead source, VARCHAR NOT NULL UNIQUE

	Catered by
	- ID, PK unsigned Int
	- Catered by, VARCHAR NOT NULL UNIQUE

	Order material offset
	- Order control no., FK
	- Material ID, FK
	- PK (Order control no., Material ID)
	- Material name, VARCHAR NOT NULL
	- Material specs, VARCHAR NULL
	- Material category, VARCHAR NULL
	- Material unit, VARCHAR NULL
	- Material remarks, VARCHAR NULL
	- Material color, VARCHAR NULL
	- Unit cost, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Quantity, Int NOT NULL DEFAULT 0

	Order prefab
	- Order control no., FK
	- Prefab ID, FK
	- PK (Order control no., Prefab ID)
	- Prefab name, VARCHAR NOT NULL
	- Prefab specs, VARCHAR NULL
	- Prefab color, VARCHAR NULL
	- Unit cost, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Quantity, unsigned Int NOT NULL DEFAULT 0

	Order prefab material
	- Order control no., FK
	- Prefab ID, FK
	- Material ID, FK
	- PK (Order control no., Prefab ID, Material ID)
	- Material name, VARCHAR NOT NULL
	- Material specs, VARCHAR NULL
	- Material category, VARCHAR NULL
	- Material unit, VARCHAR NULL
	- Material remarks, VARCHAR NULL
	- Material color, VARCHAR NULL
	- Unit cost, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Material quantity per prefab, unsigned Int NOT NULL DEFAULT 0

	Order other works
	- Order control no., FK
	- Other works ID, FK
	- PK (Order control no., Other works ID)
	- Other works name, VARCHAR NOT NULL
	- Other works category, VARCHAR NULL
	- Other works unit, VARCHAR NULL
	- Unit cost, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Quantity, unsigned Int NOT NULL DEFAULT 0

--- Payment tables ---

	Payment
	- ID, PK unsigned Int
	- Order control no., FK
	- Date paid, Date NOT NULL DEFAULT CURRENT_TIMESTAMP
	- Balance, Decimal(12,2) NOT NULL DEFAULT (Order control no. -> Grand Total)
	- Payment amount, Decimal(12,2) NOT NULL DEFAULT 0.00
	- Payment method ID, FK
	- Credit to account ID, FK
	- Deposit date, DATE NULL
	- Payment status ID, FK
	- Deposit slip, VARCHAR NULL

	Payment method
	- ID, PK unsigned Int
	- Payment method, VARCHAR NOT NULL UNIQUE

	Credit to account
	- ID, PK unsigned Int
	- Credit to account, VARCHAR NOT NULL UNIQUE

	Payment status
	- ID, PK unsigned Int
	- Payment status, VARCHAR NOT NULL UNIQUE

--- Purchase order tables ---

	Purchase order
	- PI no., PK unsigned Int
	- Batch, unsigned Int NOT NULL
	- Batch year, FK unsigned Int NOT NULL
	- UNIQUE (Batch, Batch year)
	- Date submitted, Date NULL
	- Date approved, Date NULL
	- Subtotal USD, Decimal(12,2) NULL
	- CIF USD, Decimal(12,2) NULL
	- Total USD, Decimal(12,2) NULL
	- Total PHP, Decimal(12,2) NULL
	- Customs PHP, Decimal(12,2) NULL
	- Container no., VARCHAR NULL
	- Remarks, TEXT NULL
	- Date delivered, Date NULL
	- Date arrived, Date NULL

	Purchase order prefab

	- Unit cost, Decimal(12,2) NOT NULL DEFAULT 0.00
